   -- inner join [common rows will return] 
   select empname,empsalary,empaddress,emplocation from office inner join staff on office.empid=staff.empid;
   
   -- left join [left table will return completely along with common row in right table
   select * from office left join staff on office.empid=staff.empid;
   
   -- right join [right table will return completely along with common row in left table
   select * from office right join staff on office.empid=staff.empid;
   
   -- full join [show all row from 2 table] 
    select * from office full join staff;
  
   -- crossjoin[one to many row comparison]
